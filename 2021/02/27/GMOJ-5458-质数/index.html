GMOJ 5458.质数

发表于 2021-02-27 分类于 模拟赛题目

样例解释

数据范围

题解

可以用筛法直接预处理出1~10^7中的质数，枚举每个数，算出是否可以被被两个质数相乘得到，也记录起来，然后做前缀和，输出直接前缀和输出就可以了

AC程序

#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#define N 10000000
using namespace std;
int
 n,m,a[10000001],s[10000001],v[10000001],x,y,sum[10000001];
int main()
{
    freopen("prime.in","r",stdin);
    freopen("prime.out","w",stdout);

    scanf("%d",&n);
    for(int i=2;i<=N;i++)
    {
        if(a[i]==0) s[++m]=i;
        for(int j=1;j<=m;j++)
        {
            if(i*s[j]>=N) break;
            a[i*s[j]]=1;
            if(i%s[j]==0) break;
        }
    }
    a[1]=1;
    for(int i=1;i<=N;i++)
    {
        if(a[i]==0) sum[i]=sum[i-1]+1;
        else
        {
            bool k=0;
            for(int j=1;j<=m;j++)
            {
                if(i%s[j]==0)
                {
                    if(!a[i/s[j]])
                    {
                        sum[i]=sum[i-1]+1;
                        k=1;
                    }
                    break;
                }
                if(i<s[j]*s[j]) break;
            }
            if(k==0) sum[i]=sum[i-1];
        }
    }
    for(int i=1;i<=n;i++)
    {
        scanf("%d%d",&x,&y);
        printf("%d\n",sum[y]-sum[x-1]);
    }
}
